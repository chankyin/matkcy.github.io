def HestonSimulateFT(S0, r, v0, kappa, theta, eta, rho, T, NTimeSteps): # one simulation
    t = linspace(0,T,NTimeSteps) # time points: index 0,1,...,NTimeSteps-1
    N = NTimeSteps # last time index
    dt = T/N
    v = np.zeros(N)
    S = np.zeros(N)
    v[0] = v0
    S[0] = S0
    
    Zv = norm.rvs(size=N)
    for i in range(1,N):
        v[i] = v[i-1]+kappa*dt*(theta-max(v[i-1],0))+eta*sqrt(max(v[i-1],0))*Zv[i]*sqrt(dt)
    
    Z=norm.rvs(size=N)
    Zs = np.zeros(N)
    Zs = (rho*Zv)+(sqrt(1-rho**2))*Z
        
    for i in range(1,N):
        S[i] = S[i-1]*exp((r-0.5*(max(v[i-1],0)))*dt+sqrt(max(v[i-1],0))*Zs[i]*sqrt(dt))
        
    
    
    return S
